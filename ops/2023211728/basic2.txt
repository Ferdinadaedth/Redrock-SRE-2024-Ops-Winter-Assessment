#将脚本命名为network.sh
#经查找后可以使用open vswitch来给虚拟机分配ip
#!/bin/bash
#先创建虚拟交换机
sudo ovs-vsctl add-br br0
#添加物理网络接口
sudo ovs-vsctl add-port br0 eth0
#分配ip和子网掩码
for i in {2..11}; do
    sudo ifconfig eth0:$i 192.168.5.$i netmask 255.255.255.0 up
done
#保存退出后将其变为可执行文件
chmod +x network.sh
#用crontab来定期运行该脚本
crontab -e
#进入编辑后输入
@reboot /home/network.sh
~                                                                                                                                                                                                             
~
